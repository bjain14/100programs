#include <stdio.h>
int main()
{
    int n,result=0;
    printf("Enter a number:");
    scanf("%d",&n);
    for(int i=1;i<=n/2;i++)
    {
        if(n%i==0)
          result=result+i;
    }
       
    if(result==n)
      printf("%d is a perfect number.",n);
    else
      printf("%d is not a perfect number.",n);  

    return 0;
}
